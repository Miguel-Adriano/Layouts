function showNotification(message, type = 'success') {
    const notification = document.getElementById('elegant-notification');
    const messageEl = document.getElementById('notification-message');
    const iconEl = document.getElementById('notification-icon');
    if (!notification || !messageEl || !iconEl) return;

    // Limpa o timer anterior
    clearTimeout(notificationTimeout);

    // 1. Define o ícone e a mensagem
    iconEl.innerHTML = elegantIcons[type];
    messageEl.textContent = message;

    // 2. Define a classe para a cor do ícone
    notification.classList.remove('success', 'error', 'warn');
    notification.classList.add(type);

    // 3. Ativa a animação de entrada
    notification.classList.add('show');

    // 4. Agenda a animação de saída
    notificationTimeout = setTimeout(() => {
        notification.classList.remove('show');
    }, 4000);
}
